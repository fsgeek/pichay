    36→def _write_jsonl(tmp_path: Path, name: str, records: list[dict]) -> Path:
    37→    """Write a list of dicts as JSONL and return the path."""
    38→    p = tmp_path / name
    39→    with open(p, "w", encoding="utf-8") as f:
    40→        for rec in records:
    41→            f.write(json.dumps(rec) + "\n")
    42→    return p

   120→def _make_system_blocks(text: str = SYSTEM_PROMPT_TEXT) -> list[dict]:
   121→    """Build a system prompt block list like the proxy captures."""
   122→    return [{"type": "text", "text": text}]

   125→def _make_user_msg_with_reminders(
   126→    *reminder_texts: str,
   127→    user_text: str = "Hello",
   128→) -> dict:
   129→    """Build a user message with system-reminder injections."""
   130→    content = user_text
   131→    for rt in reminder_texts:
   132→        content += f"\n<system-reminder>\n{rt}\n</system-reminder>"
   133→    return {"role": "user", "content": content}
   134→
   135→
   136→def _make_user_msg_with_block_reminders(
   137→    *reminder_texts: str,
   138→    user_text: str = "Hello",
   139→) -> dict:
   140→    """Build a user message with block-style content including reminders."""
   141→    blocks: list[dict] = [{"type": "text", "text": user_text}]
   142→    for rt in reminder_texts:
   143→        blocks.append({
   144→            "type": "text",
   145→            "text": f"<system-reminder>\n{rt}\n</system-reminder>",
   146→        })
   147→    return {"role": "user", "content": blocks}

   158→def _make_request(
   159→    system_prompt_full: list[dict] | None = None,
   160→    messages_full: list[dict] | None = None,
   161→    total_request_bytes: int = 50000,
   162→    system_prompt_bytes: int = 15000,
   163→    messages_total_bytes: int = 20000,
   164→    ts: str = "2026-03-01T12:00:00Z",
   165→) -> dict:
   166→    """Build a proxy request record with all fields the analyzer expects."""
   167→    rec: dict = {
   168→        "type": "request",
   169→        "timestamp": ts,
   170→        "model": "claude-opus-4-6",
   171→        "total_request_bytes": total_request_bytes,
   172→        "system": {"system_prompt_bytes": system_prompt_bytes},
   173→        "messages": {"messages_total_bytes": messages_total_bytes},
   174→    }
   175→    if system_prompt_full is not None:
   176→        rec["system_prompt_full"] = system_prompt_full
   177→    if messages_full is not None:
   178→        rec["messages_full"] = messages_full
   179→    return rec

   482→            messages_full=[{"role": "user", "content": "turn 2"}],
   483→            ts="2026-03-01T12:01:00Z",
   484→        )
   485→        return _write_jsonl(tmp_path, "multi.jsonl", [rec1, rec2])

   487→    def test_identical_prompts_are_static(self, tmp_path):
   488→        """When SP is identical across turns, all components are static."""
   489→        p = self._two_turn_log(
   490→            tmp_path, SYSTEM_PROMPT_TEXT, SYSTEM_PROMPT_TEXT,
   491→        )
   492→        a = analyze_system_prompts(p)
   493→
   494→        assert a.api_calls == 2
   495→        for track in a.component_tracks:
   496→            assert track.is_static, f"{track.name} should be static"
   497→            assert track.change_count == 0
   498→            assert track.turns_present == 2
   499→        assert a.dynamic_bytes == 0
   500→        assert a.static_bytes > 0

   502→    def test_changed_git_status_is_dynamic(self, tmp_path):
   503→        """When git status changes between turns, that section is dynamic."""

   512→        track_by_name = {t.name: t for t in a.component_tracks}

   524→    def test_static_wasted_bytes(self, tmp_path):
   525→        """Static re-send waste = sum of sizes[1:] for static components."""
   526→        p = self._two_turn_log(
   527→            tmp_path, SYSTEM_PROMPT_TEXT, SYSTEM_PROMPT_TEXT,
   528→        )
   529→        a = analyze_system_prompts(p)
   530→
   531→        # Every component appears twice; the second send is wasted
   532→        assert a.static_wasted_bytes > 0
   533→        # Wasted should equal roughly the same as the first turn's SP bytes
   534→        assert a.static_wasted_bytes == pytest.approx(
   535→            a.total_system_prompt_bytes / 2, rel=0.01,
   536→        )
